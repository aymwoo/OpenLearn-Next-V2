    ```typescript
    /**
     * EventBusProxy — Worker-side event subscription proxy.
     *
     * Intercepts eventBus.subscribe() and eventBus.unsubscribe() calls,
     * sends subscribe/unsubscribe messages to main thread, and dispatches
     * forwarded events to local handlers.
     *
     * Structured clone cannot transfer functions — the subId proxy pattern
     * keeps handler functions in the Worker and sends only subId + eventType
     * across the boundary. (See RESEARCH.md Common Pitfall 1)
     */
    export class EventBusProxy {
      // Map<subId, handlers[]> — each subId can have multiple handlers
      private subscriptions = new Map<string, Array<(event: any) => void>>();
      private transport: IWorkerTransport;

      constructor(transport: IWorkerTransport) {
        this.transport = transport;
      }

      /**
       * Subscribe to an event type. Generates subId, sends subscribe message,
       * tracks the handler for later dispatch when forwarded events arrive.
       */
      subscribe(eventType: string, handler: (event: any) => void): string {
        const subId = crypto.randomUUID();
        const handlers = this.subscriptions.get(subId) ?? [];
        handlers.push(handler);
        this.subscriptions.set(subId, handlers);

        this.transport.postMessage({
          type: 'subscribe' as const,
          subId,
          eventType,
        });
        return subId;
      }

      /**
       * Unsubscribe a handler. If it was the last handler for a subId,
       * sends unsubscribe message to main thread.
       */
      unsubscribe(eventType: string, handler: (event: any) => void): void {
        for (const [subId, handlers] of this.subscriptions) {
          const idx = handlers.indexOf(handler);
          if (idx !== -1) {
            handlers.splice(idx, 1);
            if (handlers.length === 0) {
              this.subscriptions.delete(subId);
              this.transport.postMessage({
                type: 'unsubscribe' as const,
                subId,
              });
            }
            break;
          }
        }
      }

      /**
       * Handle an incoming forwarded event from main thread.
       * Called by the onMessage handler in the Worker bootstrap.
       */
      handleEvent(subId: string, event: any): void {
        const handlers = this.subscriptions.get(subId);
        if (!handlers) return;
        for (const handler of handlers) {
          try { handler(event); } catch (err) {
            console.error('[EventBusProxy] Handler error:', err);
          }
        }
      }

      /**
       * Clean up all subscriptions (called when Worker is shutting down).
       * Sends unsubscribe for all active subIds.
       */
      disposeAll(): void {
        for (const [subId] of this.subscriptions) {
          this.transport.postMessage({
            type: 'unsubscribe' as const,
            subId,
          });
        }
        this.subscriptions.clear();
      }
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| EventBus <-> EventForwarder | Trusted: EventForwarder subscribes to EventBus on main thread and forwards events to Worker. |
| Worker EventBusProxy <-> Plugin handler | Worker-local: handlers run inside the Worker. EventBusProxy dispatches forwarded events to local handlers. |
| ServiceHost subscribe handler | Untrusted: Worker sends subscribe message with eventType. EventForwarder creates EventBus subscription. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-15 | Information Disclosure | event-forwarder.ts event forwarding | mitigate | EventForwarder only forwards events the Worker explicitly subscribed to. It does NOT broadcast all system events. The subscribe message includes eventType which must match an actual EventBus event type. |
| T-05-16 | Denial of Service | event-forwarder.ts subscribe flood | mitigate | Worker registry already limits active Workers to 32. Each Worker's EventForwarder tracks subscriptions per subId. No per-EventForwarder limit is needed at this point — event subscription is bounded by the EventBus itself. |
| T-05-17 | Tampering | service-proxy.ts EventBusProxy | mitigate | EventBusProxy only dispatches events to handlers that explicitly subscribed. The subId proxy pattern prevents handler function references from crossing the boundary (functions cannot be cloned). |
| T-05-18 | Tampering | service-host.ts EventForwarder dispose | mitigate | disposeEventForwarder is called before Worker termination in finally block. Even if the Worker crashes unexpectedly, the 'exit' handler in WorkerRegistry triggers cleanup. |
| T-05-19 | Elevation of Privilege | service-host.ts subscribe | accept | A Worker could theoretically subscribe to sensitive event types it shouldn't have access to. At Phase 5, there is no event-level capability check. This is documented as a gap. Phase 6 adds IEventBusService which can integrate capability checks per event type. Mitigation: current events contain no PII or secrets (lesson.created, whiteboard.element_drawn, etc.). |
