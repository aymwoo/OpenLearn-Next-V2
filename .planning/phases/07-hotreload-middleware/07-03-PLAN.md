```typescript
/** 生命周期阶段 — 中间件挂载点 */
export type LifecyclePhase =
  | 'beforeActivate'
  | 'afterActivate'
  | 'beforeDeactivate'
  | 'afterDeactivate'
  | 'beforeCommand'
  | 'afterCommand';

/** 中间件上下文 — 传递给每个中间件的不可变数据 */
export interface MiddlewareContext {
  readonly pluginId: string;
  readonly manifest: Manifest;
  readonly phase: LifecyclePhase;
  readonly timestamp: number;
}

/** 
 * 中间件函数 — 洋葱模型。
 * 
 * @param ctx - 中间件上下文
 * @param next - 调用下一个中间件或实际处理器
 * 
 * 中间件在 next() 之前做预处理，next() 之后做后处理。
 * next() 返回 Promise<void> —— await next() 等待内层完成。
 */
export type Middleware = (
  ctx: MiddlewareContext,
  next: () => Promise<void>,
) => Promise<void>;
```

在 middleware.ts 中创建 compose 函数:
```typescript
/**
 * 将中间件数组组合成洋葱管道。
 *
 * 调用顺序（3 个中间件）:
 *   middleware1.pre → middleware2.pre → middleware3.pre
 *     → handler
 *   middleware3.post ← middleware2.post ← middleware1.post
 *
 * 每个中间件被 try/catch 包裹 — 单个中间件异常不阻塞其余。
 *
 * @param middlewares - 按注册顺序排列的中间件数组
 * @returns 组合后的函数，接受 (ctx, handler)
 */
export function compose(
  middlewares: Middleware[],
): (ctx: MiddlewareContext, handler: () => Promise<void>) => Promise<void>;

/** 单次中间件调用（带错误隔离） */
export function safeInvoke(
  middleware: Middleware,
  ctx: MiddlewareContext,
  next: () => Promise<void>,
): Promise<void>;
```

compose 实现细节:
```typescript
export function compose(middlewares: Middleware[]) {
  return async (ctx: MiddlewareContext, handler: () => Promise<void>) => {
    let index = -1;
    
    const dispatch = async (i: number): Promise<void> => {
      if (i <= index) throw new Error('[Middleware] next() called multiple times');
      index = i;
      
      if (i >= middlewares.length) {
        // 所有中间件执行完毕，调用实际处理器
        return handler();
      }
      
      const middleware = middlewares[i];
      // 错误隔离: 每个中间件包裹在 try/catch
      return safeInvoke(middleware, ctx, () => dispatch(i + 1));
    };
    
    await dispatch(0);
  };
}
```

safeInvoke 实现细节:
```typescript
export async function safeInvoke(
  middleware: Middleware,
  ctx: MiddlewareContext,
  next: () => Promise<void>,
): Promise<void> {
  try {
    await middleware(ctx, next);
  } catch (err) {
    console.error(
      `[PluginHost] Middleware error in phase "${ctx.phase}" for plugin "${ctx.pluginId}":`,
      err,
    );
    // 不重新抛出 — 继续执行下一个中间件
  }
}
```

**Verification:**
- compose([])(ctx, handler) 直接调用 handler
- compose([m1, m2])(ctx, handler) 按 m1→m2→handler→m2→m1 顺序执行
- 中间件抛异常时后续中间件继续执行，handler 最终仍被调用
- 中间件调用 next() 两次时抛出错误（防止死循环）

实际上，更简洁的方式是:

```typescript
async activatePlugin(pluginId: string): Promise<void> {
  // ... 前置准备（manifest, semver, context）...

  const ctx: MiddlewareContext = {
    pluginId, manifest, phase: 'beforeActivate', timestamp: Date.now(),
  };

  // 实际激活
  const activateFn = async () => {
    // ESM load + ctx.activate() + 状态更新 + 事件发布
  };

  // 洋葱管道: beforeActivate → activate → afterActivate
  const before = this.middlewareRegistry.get('beforeActivate') ?? [];
  const after = this.middlewareRegistry.get('afterActivate') ?? [];

  const pipeline = compose([
    ...before,
    async (_ctx, next) => {
      await next(); // 执行激活
      // 激活成功后执行 afterActivate 中间件
      const afterCtx = { ...ctx, phase: 'afterActivate' as LifecyclePhase };
      const afterPipeline = compose(after);
      await afterPipeline(afterCtx, async () => {});
    },
  ]);

  await pipeline(ctx, activateFn);
}
```

**beforeCommand/afterCommand — CommandBus 集成:**

```typescript
// packages/core/command-bus/index.ts
export class CommandBus {
  private commandMiddleware: { before: Middleware[]; after: Middleware[] } = { before: [], after: [] };

  setCommandMiddleware(before: Middleware[], after: Middleware[]): void {
    this.commandMiddleware = { before, after };
  }

  async execute<T extends PlatformCommand>(command: T): Promise<any> {
    // ... existing normalization ...
    
    // beforeCommand 洋葱管道
    const cmdCtx: MiddlewareContext = {
      pluginId: command.actorId,
      manifest: { id: command.type } as any,
      phase: 'beforeCommand',
      timestamp: Date.now(),
    };
    const beforePipeline = compose(this.commandMiddleware.before);
    await beforePipeline(cmdCtx, async () => {
      // 现有 interceptor（capability + high-risk check）
      if (this.interceptor) await this.interceptor(normalizedCommand);
    });

    // Execute handler
    const result = await handler.execute(normalizedCommand);

    // afterCommand 洋葱管道
    const afterCtx = { ...cmdCtx, phase: 'afterCommand' as LifecyclePhase };
    const afterPipeline = compose(this.commandMiddleware.after);
    await afterPipeline(afterCtx, async () => {});

    return result;
  }
}
```

### Task 3: 中间件单元测试

**Files:**
- `packages/core/plugin-host/__tests__/middleware.test.ts` — 新建

**Description:**

添加 8 个测试用例:

1. **空中间件数组 — handler 直接执行**
   - compose([])(ctx, handler) → handler 被调用

2. **单中间件 — 顺序正确**
   - 中间件记录 `pre` → handler 记录 `work` → 中间件记录 `post`
   - 结果: `['pre', 'work', 'post']`

3. **多中间件 — 洋葱顺序**
   - m1, m2, m3 → 执行顺序: m1.pre → m2.pre → m3.pre → handler → m3.post → m2.post → m1.post

4. **中间件跳过 next() — handler 不被调用**
   - 中间件不调用 next() → handler 不被执行
   - 后续中间件也不执行 (next 没有被调用)

5. **中间件抛异常 — 错误隔离**
   - m1 抛出错误 → m2 继续执行 → handler 最终被调用
   - console.error 被调用一次

6. **next() 调用两次 — 抛出错误**
   - 中间件中调用 await next(); await next(); → 第二次抛 Error('next() called multiple times')

7. **registerMiddleware + activatePlugin 集成**
   - 注册 beforeActivate 中间件 → 激活插件 → 验证中间件在 activate 前执行

8. **afterActivate 中间件仅在激活成功后执行**
   - 注册 afterActivate 中间件 → 激活插件 → 验证中间件在 activate 后被调用
   - 激活失败（源码语法错误）→ afterActivate 中间件不被调用

9. **beforeCommand 中间件在命令执行前触发**
   - 注册 beforeCommand 中间件 → execute(command)
   - 验证: 中间件在 handler.execute 之前被调用

10. **afterCommand 中间件在命令执行后触发**
    - 注册 afterCommand 中间件 → execute(command)
    - 验证: 中间件在 handler.execute 之后被调用

11. **beforeCommand 中间件跳过 next() → 命令不被执行**
    - 注册 beforeCommand 中间件不调用 next()
    - 验证: handler.execute 不被调用（interceptor 也不执行）

- 11 个测试全部通过

- 测试覆盖洋葱模型所有关键路径（成功、失败、跳过、异常隔离）
