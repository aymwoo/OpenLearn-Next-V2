    1. **Keep existing default export** for backward compatibility (D-12): `export default function App() { ... }`

    2. **Add named export createMfeApp** (D-05):
    ```typescript
    import React from 'react';
    import { createRoot, Root } from 'react-dom/client';
    import type { MfeContext } from '@/src/mfe/types';

    // ── Component ────────────────────────────────────────────────────

    export default function App() {
      return <div>Whiteboard MFE</div>;
    }

    // ── Lifecycle Factory ────────────────────────────────────────────

    export function createMfeApp(ctx: MfeContext) {
      console.log('[mfe-whiteboard] Initialized with context:', ctx);

      return {
        mount: async (container: HTMLElement, props?: Record<string, any>) => {
          const root = createRoot(container);
          root.render(<App {...props} />);

          return {
            unmount: async () => {
              root.unmount();
            },
            update: async (newProps: Record<string, any>) => {
              root.render(<App {...newProps} />);
            },
          };
        },
        styles: [] as string[],
      };
    }
    ```

    3. **Type import path:** Use `@/src/mfe/types` if the remote package's vite.config.ts or tsconfig resolves the `@` alias. If not, use a relative path: `../../../src/mfe/types`. The executor should test first with `@/src/mfe/types` and fall back to relative path if compilation fails.

    4. **The existing exposes config does NOT need to change** — `'./App': './src/App.tsx'` will export the module that contains both the default `App` component AND the named `createMfeApp` function. `loadRemote('mfe_whiteboard/App')` returns `{ default: App, createMfeApp }`.

    5. **No dynamic shared dependency changes needed.** The remote already has `react` and `react-dom` in shared config with `singleton: true` in its vite.config.ts.
